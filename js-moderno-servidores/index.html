JS05

Callbacks · forEach · for in/of · try/catch · throw

Callback

Función pasada como argumento y ejecutada más tarde, base del modelo asíncrono de Node.js.

forEach / for in / for of

Tres formas de iterar: forEach sobre arrays, for..in sobre claves de objetos, for..of sobre valores iterables.

try / catch / throw

Manejo estructurado de errores: try ejecuta, catch captura, throw lanza excepciones personalizadas.

js05_fundamentos.js 
// ── CALLBACK ─────────────────────────────────────────────────────
// Un callback es una función que se pasa como argumento a otra
// función para ser ejecutada cuando se complete alguna operación.
function leerArchivo(ruta, callback) {
  // Simulamos lectura asíncrona de un archivo en servidor
  setTimeout(() => {
    const datos = { usuario: "admin", rol: "superuser" };
    callback(null, datos);   // patrón Node: primer arg = error
  }, 500);
}

leerArchivo("/etc/config.json", (err, data) => {
  if (err) return console.error(err);
  console.log("Usuario:", data.usuario);
});

// ── FOREACH ──────────────────────────────────────────────────────
// Itera cada elemento de un array ejecutando la función dada.
// No retorna nuevo array (diferente a map).
const rutas = ["/api/users", "/api/posts", "/api/auth"];
rutas.forEach((ruta, i) => {
  console.log(`Registrando ruta [${i}]: ${ruta}`);
});

// ── FOR IN ───────────────────────────────────────────────────────
// Itera las CLAVES enumerables de un objeto.
// Ideal para inspeccionar propiedades de configuración.
const config = { host: "localhost", port: 3000, env: "dev" };
for (const clave in config) {
  console.log(`  ${clave} = ${config[clave]}`);
}

// ── FOR OF ───────────────────────────────────────────────────────
// Itera los VALORES de cualquier objeto iterable
// (arrays, strings, Map, Set, generadores, etc.).
const middlewares = new Set(["cors", "helmet", "rateLimit"]);
for (const mw of middlewares) {
  console.log(`Cargando middleware: ${mw}`);
}

// ── TRY / CATCH / THROW ──────────────────────────────────────────
// try: bloque que puede lanzar errores
// catch: captura el error para manejo controlado
// throw: lanza una excepción personalizada
// finally: siempre se ejecuta (limpieza de recursos)
class HttpError extends Error {
  constructor(status, message) {
    super(message);
    this.status = status;
    this.name = "HttpError";
  }
}

function validarToken(token) {
  try {
    if (!token) throw new HttpError(401, "Token requerido");
    if (token !== "secreto123") throw new HttpError(403, "Token inválido");
    return { ok: true, usuario: "admin" };
  } catch (err) {
    console.error(`[${err.status}] ${err.message}`);
    return { ok: false, error: err.message };
  } finally {
    console.log("Validación completada");  // siempre se ejecuta
  }
}
ES06

let · const · Template Strings · Arrow Functions · Clases · Map · Filter · Spread · Rest · Promises

es06_moderno.js 
// ── LET & CONST ──────────────────────────────────────────────────
// let: variable reasignable con ámbito de bloque (reemplaza var)
// const: referencia constante (el objeto puede mutar, no la ref)
let   intentos = 0;
const MAX_INTENTOS = 3;
const servidor     = { host: "0.0.0.0", port: 8080 };

// ── TEMPLATE STRINGS ─────────────────────────────────────────────
// Cadenas con interpolación ${} y saltos de línea literales.
const bienvenida = `Servidor corriendo en
  http://${servidor.host}:${servidor.port}
  Modo: ${process.env.NODE_ENV ?? "development"}`;

// ── ARROW FUNCTION ───────────────────────────────────────────────
// Sintaxis compacta; no tiene su propio 'this'.
// Ideal para callbacks y funciones de utilidad.
const sumar       = (a, b) => a + b;
const esAdmin     = usuario => usuario.rol === "admin";
const saludar     = nombre => ({ mensaje: `Hola ${nombre}` });

// ── CLASS ────────────────────────────────────────────────────────
// Sintaxis de clase sobre el sistema de prototipos de JS.
// Permite herencia, constructores y métodos de instancia.
class Repositorio {
  constructor(nombre) {
    this.nombre = nombre;
    this.items  = [];
  }
  guardar(item) { this.items.push(item); return this; }  // fluent API
  buscar(id)  { return this.items.find(i => i.id === id); }
  static crear(nombre) { return new Repositorio(nombre); }
}

class RepositorioUsuario extends Repositorio {
  constructor() { super("usuarios"); }
  admins() { return this.items.filter(esAdmin); }
}

// ── MAP & FILTER ─────────────────────────────────────────────────
// map:    transforma cada elemento → nuevo array del mismo tamaño
// filter: selecciona elementos que pasan el predicado
const precios = [100, 250, 50, 430, 89];
const conIva   = precios.map(p => p * 1.19);              // aplica IVA 19%
const costosos = precios.filter(p => p > 100);            // [250, 430]

// ── SPREAD & REST ────────────────────────────────────────────────
// spread (...): expande un iterable en elementos individuales
// rest   (...): agrupa argumentos restantes en un array
const base  = { timeout: 5000, retries: 3 };
const prod  = { ...base, ssl: true, timeout: 10000 };  // merge + override

function log(nivel, ...mensajes) {                       // rest parameter
  mensajes.forEach(m => console.log(`[${nivel}] ${m}`));
}

// ── PROMISES ─────────────────────────────────────────────────────
// Una Promise representa un valor futuro: pending → fulfilled | rejected
// .then() para éxito, .catch() para error, .finally() siempre.
function conectarDB(url) {
  return new Promise((resolve, reject) => {
    setTimeout(() => {
      if (url.startsWith("mongodb"))
        resolve({ ok: true, latencia: "12ms" });
      else
        reject(new Error("Protocolo no soportado"));
    }, 200);
  });
}

conectarDB("mongodb://localhost:27017")
  .then(r => console.log("Conectado en", r.latencia))
  .catch(e => console.error(e.message));
ES07 — ES08

includes · ** · Object.values/entries · String Padding · Async/Await

es07_es08.js 
// ── ES07: Array.includes ──────────────────────────────────────────
// Comprueba si un valor existe en el array (más legible que indexOf).
// También funciona con NaN, a diferencia de indexOf.
const metodosPermitidos = ["GET", "POST", "PUT", "DELETE"];
const validarMetodo = m => metodosPermitidos.includes(m.toUpperCase());

// ── ES07: Operador exponencial (**) ───────────────────────────────
// Equivalente a Math.pow(base, exp). Más legible en criptografía,
// algoritmos y cálculos científicos en el servidor.
const calcHash = (base, exp, mod) => (base ** exp) % mod;
const kb  = 2 ** 10;   // 1024
const mb  = 2 ** 20;   // 1 048 576

// ── ES08: Object.values y Object.entries ─────────────────────────
// values():  retorna array de valores del objeto
// entries(): retorna array de pares [clave, valor]
const rutas = {
  home:    "/inicio",
  users:   "/api/usuarios",
  health:  "/health",   // trailing comma ← ES08
};

const urls  = Object.values(rutas);     // ["/inicio", "/api/usuarios", "/health"]
const pares = Object.entries(rutas);    // [["home","/inicio"], ...]

pares.forEach(([nombre, ruta]) => {   // desestructuración
  console.log(`  ${nombre.padEnd(8)} → ${ruta}`);
});

// ── ES08: String Padding ──────────────────────────────────────────
// padStart(): rellena al inicio hasta alcanzar longitud dada
// padEnd():   rellena al final — útil para logs alineados en servidor
const formatoLog = (id, estado) =>
  `REQ-${String(id).padStart(5, "0")} [${estado.padEnd(7)}]`;
console.log(formatoLog(42, "200 OK"));   // "REQ-00042 [200 OK ]"

// ── ES08: Async / Await ───────────────────────────────────────────
// Sintaxis sobre Promises que hace el código asíncrono lucir síncrono.
// async: marca la función como asíncrona (retorna Promise implícita)
// await: suspende la ejecución hasta que la Promise resuelva.
async function iniciarServidor() {
  try {
    const db    = await conectarDB("mongodb://localhost:27017");
    const cache = await conectarRedis();
    console.log("✓ DB y caché listos");
  } catch (err) {
    console.error("Error de inicio:", err.message);
    process.exit(1);
  }
}
ES09 · ES10 · ES11

Rest/Spread extendido · Promise.finally · BigInt · flatMap · Dynamic Import · allSettled · ?. · ??

es09_es10_es11.js 
// ── ES09: Rest/Spread en objetos ──────────────────────────────────
// En ES6 solo funcionaba en arrays; ahora también en objetos.
// Permite extraer o fusionar propiedades elegantemente.
const peticion = { id: 1, cuerpo: {nombre:"Ana"}, token: "abc", ip: "1.2.3.4" };
const { token, ip, ...datosSeguros } = peticion;  // excluye datos sensibles

// ── ES09: Promise.finally ─────────────────────────────────────────
// Se ejecuta siempre (éxito o error). Ideal para cerrar conexiones.
let conexion;
abrirConexion()
  .then(c => { conexion = c; return c.query("SELECT 1"); })
  .catch(err => console.error(err))
  .finally(() => conexion?.cerrar());           // siempre libera el recurso

// ── ES10: BigInt ──────────────────────────────────────────────────
// Tipo para enteros mayores a Number.MAX_SAFE_INTEGER (2^53-1).
// Esencial en criptografía, IDs de 64 bits y finanzas de alta precisión.
const idTransaccion = 9007199254740992n;     // sufijo 'n' = BigInt
const siguiente     = idTransaccion + 1n;

// ── ES10: trimStart / trimEnd ─────────────────────────────────────
// Elimina espacios solo al inicio o al final de una cadena.
// Útil al sanitizar headers HTTP o inputs del usuario.
const header  = "  Bearer token123  ";
const limpio  = header.trimStart().trimEnd();  // "Bearer token123"

// ── ES10: Array.flatMap ───────────────────────────────────────────
// map() + flat(1) en un solo paso. Ideal para transformar arrays
// donde cada elemento produce cero o más resultados.
const usuarios = [
  { nombre: "Ana",   roles: ["admin", "editor"] },
  { nombre: "Luis",  roles: ["viewer"]         },
];
const todosLosRoles = usuarios.flatMap(u => u.roles);
// ["admin", "editor", "viewer"]

// ── ES10: Dynamic import ──────────────────────────────────────────
// Carga módulos bajo demanda (lazy loading). Reduce el arranque
// del servidor cargando plugins/drivers solo cuando se necesitan.
async function cargarPlugin(nombre) {
  const plugin = await import(`./plugins/${nombre}.js`);  // import dinámico
  return plugin.default();
}

// ── ES10: Object.fromEntries ──────────────────────────────────────
// Inverso de Object.entries(): construye objeto desde pares [k, v].
// Perfecto para transformar Map o parámetros de URL en objetos.
const queryParams = new URLSearchParams("page=1&limit=20&sort=asc");
const filtros     = Object.fromEntries(queryParams);
// { page: "1", limit: "20", sort: "asc" }

// ── ES10: Optional Catch Binding ──────────────────────────────────
// El parámetro del catch ya no es obligatorio si no se usa el error.
try {
  JSON.parse("texto inválido");
} catch {                          // ← sin (err) cuando no se necesita
  console.log("JSON inválido recibido");
}

// ── ES10: globalThis ─────────────────────────────────────────────
// Referencia unificada al objeto global en cualquier entorno:
// window (browser), global (Node.js), self (Worker).
globalThis.appVersion = "2.0.0";     // funciona en browser Y en Node

// ── ES10: Private Class Variables (#) ────────────────────────────
// Campos privados verdaderos con # — no accesibles desde afuera.
class ServicioAuth {
  #secretJwt;          // propiedad privada real (no convención _)
  #intentosFallidos = 0;

  constructor(secret) {
    this.#secretJwt = secret;
  }

  login(user, pass) {
    if (this.#intentosFallidos >= 5)
      throw new Error("Cuenta bloqueada");
    this.#intentosFallidos++;
    // this.#secretJwt accesible aquí pero no fuera de la clase
  }
}

// ── ES11: Promise.allSettled ──────────────────────────────────────
// Espera TODAS las promesas (fulfilled o rejected).
// A diferencia de Promise.all, no se cancela si una falla.
const servicios = [pingDB(), pingCache(), pingStorage()];
const resultados = await Promise.allSettled(servicios);

resultados.forEach(({status, value, reason}) => {
  if (status === "fulfilled") console.log("✓", value);
  else                          console.warn("✗", reason);
});

// ── ES11: Nullish Coalescing (??) ─────────────────────────────────
// Retorna el lado derecho solo si el izquierdo es null o undefined.
// Diferencia clave: 0, "" y false pasan como valores válidos.
const cfg   = { port: 0, debug: false };
const port  = cfg.port  ?? 3000;    // → 0  (0 no es null/undefined)
const debug = cfg.debug ?? true;    // → false (false no es null/undefined)
const host  = cfg.host  ?? "0.0.0.0"; // → "0.0.0.0" (host es undefined)

// ── ES11: Optional Chaining (?.) ──────────────────────────────────
// Accede a propiedades anidadas sin lanzar error si algún nivel
// es null/undefined. Regresa undefined en vez de explotar.
const usuario = await buscarUsuario(id);
const ciudad  = usuario?.perfil?.direccion?.ciudad;  // seguro aunque perfil sea null
const perm    = usuario?.getPermisos?.();            // llamada segura a método
TypeScript

Tipado estático · Interfaces · Genéricos · Decoradores · Utility Types

TypeScript es un superset tipado de JavaScript que compila a JS puro. En el servidor añade seguridad de tipos en tiempo de compilación, autocompletado y refactoring seguro.
servidor.ts 
// ── TIPOS BÁSICOS ─────────────────────────────────────────────────
// TypeScript infiere tipos o los acepta explícitamente.
// Ventaja: los errores se detectan antes de ejecutar el código.
let   puerto: number  = 8080;
const entorno: string  = "producción";
let   activo: boolean = true;

// ── INTERFACES ────────────────────────────────────────────────────
// Define la forma de un objeto (contrato de datos).
// Los ? marcan propiedades opcionales.
interface Usuario {
  id:         number;
  nombre:     string;
  email:      string;
  rol:        "admin" | "editor" | "viewer";  // union type
  ultimoLogin?: Date;                           // opcional
}

interface Respuesta<T> {          // genérico: T se define al usar la interfaz
  ok:      boolean;
  datos?:  T;
  error?:  string;
  meta?:   { total: number; pagina: number };
}

// ── GENÉRICOS ────────────────────────────────────────────────────
// Permiten escribir funciones y clases que funcionan con cualquier
// tipo manteniendo la seguridad de tipos (como templates en C++).
async function buscar<T>(
  modelo: string,
  id:     number
): Promise<Respuesta<T>> {
  const dato = await db.findById<T>(modelo, id);
  return { ok: true, datos: dato };
}
const res = await buscar<Usuario>("usuarios", 42);  // res.datos es Usuario

// ── UTILITY TYPES ────────────────────────────────────────────────
// TypeScript incluye tipos de utilidad para transformar interfaces.
type UsuarioPublico  = Omit<Usuario, "email">;          // sin email
type ActualizarUser  = Partial<Pick<Usuario, "nombre"|"rol">>;  // solo esos campos, opcionales
type ConfigReadonly = Readonly<{host:string;port:number}>;  // inmutable

// ── DECORADORES ──────────────────────────────────────────────────
// Funciones que anotan/modifican clases, métodos o propiedades.
// Ampliamente usados en NestJS, TypeORM, Angular.
function Log(target: any, key: string, desc: PropertyDescriptor) {
  const orig = desc.value;
  desc.value = function(...args: any[]) {
    console.log(`→ ${key}(${args.join(",")})`);
    return orig.apply(this, args);
  };
}

class UserService {
  @Log                    // decorador de método — registra cada llamada
  crearUsuario(data: ActualizarUser) {
    return { id: Date.now(), ...data };
  }
}
RxJS

Observables · Operadores · pipe · Subjects · Manejo de errores reactivo

RxJS (Reactive Extensions for JavaScript) implementa la programación reactiva usando Observables: flujos de datos asíncronos que se pueden transformar, combinar y controlar con operadores.
rxjs_server.ts 
import {
  Observable, Subject, BehaviorSubject,
  from, of, interval, fromEvent, merge, combineLatest,
} from "rxjs";
import {
  map, filter, switchMap, mergeMap, concatMap,
  catchError, retry, debounceTime, distinctUntilChanged,
  takeUntil, tap, share,
} from "rxjs/operators";

// ── OBSERVABLE ───────────────────────────────────────────────────
// Un Observable es un productor lazy de valores a lo largo del tiempo.
// Solo emite cuando alguien se suscribe (pull vs push).
const contador$ = new Observable<number>(subscriber => {
  let n = 0;
  const id = setInterval(() => {
    subscriber.next(n++);          // emite valor
    if (n > 5) subscriber.complete();  // señal de fin
  }, 1000);
  return () => clearInterval(id);     // teardown al desuscribir
});

// ── PIPE Y OPERADORES ────────────────────────────────────────────
// pipe() encadena operadores que transforman el flujo de datos.
// Cada operador retorna un nuevo Observable (inmutabilidad).
const pares$ = contador$.pipe(
  filter(n => n % 2 === 0),          // solo pares
  map(n => n * 10),                  // multiplica por 10
  tap(n => console.log("DEBUG:",n)),   // efecto de lado sin mutar
);

// ── SUSCRIPCIÓN ──────────────────────────────────────────────────
// El Observable no hace nada hasta subscribe().
// Recibe tres callbacks: next, error, complete.
const sub = pares$.subscribe({
  next:     valor => console.log("Valor:", valor),
  error:    err   => console.error("Error:", err),
  complete: ()     => console.log("Completado"),
});
setTimeout(() => sub.unsubscribe(), 3000);  // evita memory leaks

// ── SUBJECT ──────────────────────────────────────────────────────
// Un Subject es Observable y Observer a la vez:
// permite emitir valores imperativamente (ej: eventos del servidor).
const eventos$ = new Subject<string>();
eventos$.subscribe(e => console.log("Evento:", e));
eventos$.next("usuario.creado");
eventos$.next("pago.procesado");

// ── SWITCHMAP ────────────────────────────────────────────────────
// Cancela la petición anterior cuando llega un nuevo valor.
// Perfecto para búsquedas en tiempo real (evita race conditions).
const busqueda$ = new Subject<string>();
busqueda$.pipe(
  debounceTime(300),             // espera 300ms de silencio
  distinctUntilChanged(),      // ignora si el valor no cambió
  switchMap(q =>               // cancela llamada anterior
    from(fetch(`/api/buscar?q=${q}`).then(r => r.json()))
      .pipe(catchError(() => of([])))  // retorna array vacío si falla
  ),
).subscribe(resultados => console.log(resultados));

// ── RETRY & CATCHERROR ────────────────────────────────────────────
// retry(n): reintenta automáticamente N veces ante error
// catchError: captura el error y puede retornar un Observable de reemplazo
const datos$ = from(fetch("/api/critico")).pipe(
  retry(3),                                    // 3 reintentos automáticos
  catchError(err => {
    console.error("Fallback activado:", err.message);
    return of({ fallback: true, datos: [] });  // valor de respaldo
  }),
);

// ── BEHAVIORSUBJECT ───────────────────────────────────────────────
// Como Subject pero guarda el último valor y lo emite al suscribirse.
// Ideal para estado global del servidor (config en caliente, health).
const estado$ = new BehaviorSubject({ online: true, peticiones: 0 });
console.log("Estado actual:", estado$.value);    // acceso síncrono
estado$.next({ online: true, peticiones: 1 });
